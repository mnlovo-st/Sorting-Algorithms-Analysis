const int SIZE = 100000;


bool readFromFile(const string& filename, int arr[], int size)
{
    ifstream inFile(filename);
    if (!inFile)
    {
        cerr << "Error: Could not open file: " << filename << endl;
        return false;
    }
    for (int i = 0; i < size; i++)
        inFile >> arr[i];
    inFile.close();
    return true;
}


void copyArray(int src[], int dst[], int size)
{
    for (int i = 0; i < size; i++)
        dst[i] = src[i];
}



void BubbleSortV1(int arr[], int size)
{
    for (int pass = 0; pass < size - 1; pass++)
    {
        for (int j = 0; j < size - pass - 1; j++)
        {
            if (arr[j] > arr[j + 1])
            {
              
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}


void BubbleSortV2(int arr[], int size)
{
    bool swapped;
    for (int pass = 0; pass < size - 1; pass++)
    {
        swapped = false;
        for (int j = 0; j < size - pass - 1; j++)
        {
            if (arr[j] > arr[j + 1])
            {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
                swapped = true;
            }
        }
        if (!swapped)
            break;
    }
}

void SelectionSortV1(int arr[], int size)
{
    for (int i = 0; i < size - 1; i++)
    {
        int minIndex = i;
        for (int j = i + 1; j < size; j++)
        {
            if (arr[j] < arr[minIndex])
                minIndex = j;
        }
        if (minIndex != i)
        {
            int temp = arr[i];
            arr[i] = arr[minIndex];
            arr[minIndex] = temp;
        }
    }
}

void SelectionSortV2(int arr[], int size)
{
    int left = 0;
    int right = size - 1;

    while (left < right)
    {
        int minIndex = left;
        int maxIndex = left;

        for (int j = left; j <= right; j++)
        {
            if (arr[j] < arr[minIndex]) minIndex = j;
            if (arr[j] > arr[maxIndex]) maxIndex = j;
        }

        if (minIndex != left)
        {
            int temp = arr[left];
            arr[left] = arr[minIndex];
            arr[minIndex] = temp;

            if (maxIndex == left)
                maxIndex = minIndex;
        }

        if (maxIndex != right)
        {
            int temp = arr[right];
            arr[right] = arr[maxIndex];
            arr[maxIndex] = temp;
        }

        left++;
        right--;
    }
}

void InsertionSortV1(int arr[], int size)
{
    for (int i = 1; i < size; i++)
    {
        int key = arr[i];
        int j = i - 1;

        while (j >= 0 && arr[j] > key)
        {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
    }
}

void InsertionSortV2(int arr[], int size)
{
    for (int i = 1; i < size; i++)
    {
        int key = arr[i];

        int lo = 0, hi = i - 1, pos = i;
        while (lo <= hi)
        {
            int mid = lo + (hi - lo) / 2;
            if (arr[mid] > key)
            {
                pos = mid;
                hi = mid - 1;
            }
            else
            {
                lo = mid + 1;
            }
        }

        for (int j = i - 1; j >= pos; j--)
            arr[j + 1] = arr[j];

        arr[pos] = key;
    }
}

void merge(int arr[], int left, int mid, int right)
{
    int n1 = mid - left + 1;
    int n2 = right - mid;

    int* L = new int[n1];
    int* R = new int[n2];

    for (int i = 0; i < n1; i++) L[i] = arr[left + i];
    for (int j = 0; j < n2; j++) R[j] = arr[mid + 1 + j];

    int i = 0, j = 0, k = left;
    while (i < n1 && j < n2)
    {
        if (L[i] <= R[j]) arr[k++] = L[i++];
        else               arr[k++] = R[j++];
    }
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];

    delete[] L;
    delete[] R;
}

void MergeSortV1(int arr[], int left, int right)
{
    if (left >= right) return;          

    int mid = left + (right - left) / 2;
    MergeSortV1(arr, left, mid);
    MergeSortV1(arr, mid + 1, right);
    merge(arr, left, mid, right);
}

void MergeSortV1(int arr[], int size)
{
    MergeSortV1(arr, 0, size - 1);
}

const int THRESHOLD = 32;

void insertionSortSubarray(int arr[], int left, int right)
{
    for (int i = left + 1; i <= right; i++)
    {
        int key = arr[i];
        int j = i - 1;
        while (j >= left && arr[j] > key)
        {
            arr[j + 1] = arr[j];
            j--;
        }
        arr[j + 1] = key;
    }
}

void MergeSortV2(int arr[], int left, int right)
{
    if (right - left < THRESHOLD)
    {
        insertionSortSubarray(arr, left, right);
        return;
    }

    int mid = left + (right - left) / 2;
    MergeSortV2(arr, left, mid);
    MergeSortV2(arr, mid + 1, right);
    merge(arr, left, mid, right);
}

void MergeSortV2(int arr[], int size)
{
    MergeSortV2(arr, 0, size - 1);
}

 
double measureTime(void (*sortFunc)(int[], int), int src[], int work[], int size)
{
    copyArray(src, work, size);       

    clock_t start = clock();
    sortFunc(work, size);
    clock_t end = clock();

    return double(end - start) / CLOCKS_PER_SEC;
}

double measureTimeMerge(void (*sortFunc)(int[], int), int src[], int work[], int size)
{
    return measureTime(sortFunc, src, work, size);
}
