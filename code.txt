// FILE STUFF

// reads numbers from file into array
bool loadFile(const string& fileName, int arr[], int n)
{
    ifstream file(fileName);

    if (!file) {
        cerr << "uhh couldn't open file: " << fileName << endl;
        return false;
    }

    for (int i = 0; i < n; i++) {
        file >> arr[i];   // assuming file is clean... hopefully
    }

    file.close(); // probably unnecessary but I like being explicit
    return true;
}

// simple copy I know there's memcpy but this is clearer tbh
void copyArr(int a[], int b[], int n)
{
    for (int i = 0; i < n; i++)
        b[i] = a[i];
}


// BUBBLE SORTS


void bubble_basic(int arr[], int n)
{
    // classic bubble sort, no shortcuts
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {

            if (arr[j] > arr[j + 1]) {
                int tmp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = tmp;
            }

        }
    }
}

void bubble_opt(int arr[], int n)
{
    bool didSwap = true; // assume worst case first

    for (int i = 0; i < n - 1 && didSwap; i++) {

        didSwap = false;

        for (int j = 0; j < n - i - 1; j++) {

            if (arr[j] > arr[j + 1]) {
                int t = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = t;
                didSwap = true;
            }
        }

        // if nothing swapped, we’re done early
    }
}


// SELECTION SORTS


void selection_basic(int arr[], int n)
{
    for (int i = 0; i < n - 1; i++) {

        int minIdx = i;

        for (int j = i + 1; j < n; j++) {
            if (arr[j] < arr[minIdx]) {
                minIdx = j;
            }
        }

        // swap if needed
        if (minIdx != i) {
            int temp = arr[i];
            arr[i] = arr[minIdx];
            arr[minIdx] = temp;
        }
    }
}

// kinda messy but works
void selection_double(int arr[], int n)
{
    int left = 0;
    int right = n - 1;

    while (left < right) {

        int minI = left;
        int maxI = left;

        for (int k = left; k <= right; k++) {
            if (arr[k] < arr[minI]) minI = k;
            if (arr[k] > arr[maxI]) maxI = k;
        }

        // move min
        int temp1 = arr[left];
        arr[left] = arr[minI];
        arr[minI] = temp1;

        // careful: max might have moved
        if (maxI == left) {
            maxI = minI;
        }
        // move max
        int temp2 = arr[right];
        arr[right] = arr[maxI];
        arr[maxI] = temp2;

        left++;
        right--;
    }
}


// INSERTION SORTS


void insertion_basic(int arr[], int n)
{
    for (int i = 1; i < n; i++) {

        int val = arr[i];
        int j = i - 1;

        while (j >= 0 && arr[j] > val) {
            arr[j + 1] = arr[j];
            j--;
        }

        arr[j + 1] = val;
    }
}

// binary search version honestly overkill but assignment said so
void insertion_binary(int arr[], int n)
{
    for (int i = 1; i < n; i++) {

        int key = arr[i];

        int lo = 0;
        int hi = i - 1;
        int pos = i;

        while (lo <= hi) {
            int mid = (lo + hi) / 2; // not worrying about overflow here

            if (arr[mid] > key) {
                pos = mid;
                hi = mid - 1;
            }
            else {
                lo = mid + 1;
            }
        }

        // shift stuff manually
        for (int j = i - 1; j >= pos; j--) {
            arr[j + 1] = arr[j];
        }

        arr[pos] = key;
    }
}


// MERGE SORT


// merge helper
void mergeStuff(int arr[], int l, int m, int r)
{
    int size1 = m - l + 1;
    int size2 = r - m;

    int* A = new int[size1];
    int* B = new int[size2];

    // copy data
    for (int i = 0; i < size1; i++) A[i] = arr[l + i];
    for (int j = 0; j < size2; j++) B[j] = arr[m + 1 + j];

    int i = 0, j = 0, k = l;

    while (i < size1 && j < size2) {
        if (A[i] <= B[j]) arr[k++] = A[i++];
        else arr[k++] = B[j++];
    }

    while (i < size1) arr[k++] = A[i++];
    while (j < size2) arr[k++] = B[j++];

    delete[] A;
    delete[] B;
}

void merge_basic(int arr[], int l, int r)
{
    if (l >= r) return;

    int mid = (l + r) / 2;

    merge_basic(arr, l, mid);
    merge_basic(arr, mid + 1, r);

    mergeStuff(arr, l, mid, r);
}

void merge_basic(int arr[], int n)
{
    merge_basic(arr, 0, n - 1);
}

// hybrid using insertion sort for small chunks
const int CUT = 32;

void insertion_small(int arr[], int l, int r)
{
    for (int i = l + 1; i <= r; i++) {
        int val = arr[i];
        int j = i - 1;

        while (j >= l && arr[j] > val) {
            arr[j + 1] = arr[j];
            j--;
        }

        arr[j + 1] = val;
    }
}

void merge_hybrid(int arr[], int l, int r)
{
    if (r - l < CUT) {
        insertion_small(arr, l, r);
        return;
    }

    int mid = (l + r) / 2;

    merge_hybrid(arr, l, mid);
    merge_hybrid(arr, mid + 1, r);

    mergeStuff(arr, l, mid, r);
}

void merge_hybrid(int arr[], int n)
{
    merge_hybrid(arr, 0, n - 1);
}


// TIMING


double timeIt(void (*func)(int[], int), int src[], int tmp[], int n)
{
    copyArr(src, tmp, n);

    clock_t start = clock();

    func(tmp, n);

    clock_t end = clock();

    return (double)(end - start) / CLOCKS_PER_SEC;
}
